
1. TypeScript class with a constructor and a function
#
class Person {

  name: string;

  age: number;

  constructor(name: string, age: number) {

    this.name = name;

    this.age = age;

  }

  sayHello() {

    console.log(`Hello, my name is ${this.name} and I'm ${this.age} years old.`);

  }

}

// Usage

const person = new Person('John', 30);

person.sayHello();
